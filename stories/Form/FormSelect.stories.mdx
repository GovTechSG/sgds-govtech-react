# ` <Form.Select />`

Customize the native `<select>`s with custom CSS that changes the elementâ€™s initial appearance.

export const TemplateWithLabel = (label, size) => {
  return (args) => (
    <Form.Group>
      <Form.Label htmlFor="defaultSelect">{label}</Form.Label>
      <FormSelect
        size={size}
        aria-label="Default select example"
        id="defaultSelect"
      >
        <option>Open this select menu</option>
        <option value="1">One</option>
        <option value="2">Two</option>
        <option value="3">Three</option>
      </FormSelect>
    </Form.Group>
  );
};

export const DefaultSelect = TemplateWithLabel('Default select menu');
export const LargeSelect = TemplateWithLabel('Large select menu', 'lg');
export const SmallSelect = TemplateWithLabel('Small select menu', 'sm');
