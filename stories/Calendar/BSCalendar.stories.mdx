import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';
import { BSCalendar } from '../../src/components/index';
import { useState } from 'react';


<Meta title="MDX/BSCalendar" component={BSCalendar} />

export const BSCalendarTemplate = () => {
  const [value, setValue] =useState(new Date().toISOString())
  const handleChange = (value, formattedValue ) => {
    setValue({value , formattedValue})
  }
    return <BSCalendar value={value} onChange={handleChange}/>
}

<Canvas>
  <Story name="Default">
      {BSCalendarTemplate.bind({})}
  </Story>
</Canvas>
