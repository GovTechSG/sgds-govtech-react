import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';
import { DatePicker, Row, Col } from '../../src/index';
import { useState } from 'react';

<Meta title="Components/DatePicker" component={DatePicker} />

export const DatePickerTemplate = (args) => {
  const [value, setValue] = useState(new Date().toISOString());
  const handleChange = (value, formattedValue) => {
    setValue({ value, formattedValue });
  };
  return (
      <div style={{width: '500px'}}>
        <DatePicker value={value} onChange={handleChange} {...args}/>
      </div>
  );
};

<Canvas>
  <Story name="Default">{DatePickerTemplate.bind({})}</Story>
</Canvas>
