  const date = new Date("2020-01-01")
  const initial = { start: new Date("2020-01-01"), end: new Date("2020-01-02") };
  const [value, setValue] = useState(initial);
  const handleChange = (value) => {
    setValue(value);
  };
  const minDate = new Date(2020, 10, 11, 12, 0, 0, 0).toISOString();
  const maxDate = new Date(2020, 10, 13, 12, 0, 0, 0).toISOString();
  return (
    <div style={{ width: '500px' }}>
      <DatePicker initialValue={initial} displayDate={date} onChange={handleChange} {...args} />
    </div>
  );
};
